InnerImg 格式互转

Base64 / HTML / SVG 核心互转在浏览器本地完成；下载时选择 JPEG、WebP、AVIF 等格式或「格式互转」页签需通过本站 API（Sharp）处理，部署时请一并启动后端服务。

Base64 与图片

支持带或不带 data:image/...;base64, 前缀；可拖拽图片文件。
解码或上传后显示

HTML 与图片

HTML → 图片：含 data:image/... 的 <img> 时按原图尺寸导出 PNG（单边最大 4096）；其它 HTML 使用内置画布 800×600 做 SVG 渲染。上传图片后会自动生成 HTML 并预览。
渲染结果或生成的 HTML

SVG 与图片

SVG → 栅格图：导出 PNG；图片 → SVG：生成包裹位图的 SVG（矢量追踪需专业工具）。
渲染或转换后显示

图片格式互转

POST /api/image/to-base64（multipart，字段 file）

Node.js 
const BASE = "http://img-convert.cs.icbc";
const fs = require("fs");
const { Blob } = require("buffer");

(async () => {
  const buf = fs.readFileSync("./photo.png");
  const blob = new Blob([buf], { type: "image/png" });
  const fd = new FormData();
  fd.append("file", blob, "photo.png");
  const res = await fetch(BASE + "/api/image/to-base64", { method: "POST", body: fd });
  console.log(await res.json());
})();
curl 
BASE=http://img-convert.cs.icbc
curl -sS -F "file=@./photo.png" "$BASE/api/image/to-base64"
Python 
import requests

BASE = "http://img-convert.cs.icbc"
with open("photo.png", "rb") as f:
    r = requests.post(f"{BASE}/api/image/to-base64", files={"file": f})
print(r.json())

POST /api/base64/to-image（JSON：dataUrl 或 base64，响应为图片二进制）

Node.js 
const BASE = "http://img-convert.cs.icbc";
const fs = require("fs");

(async () => {
  const res = await fetch(BASE + "/api/base64/to-image", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      dataUrl: "data:image/png;base64,iVBORw0KGgoAAAANS...",
    }),
  });
  fs.writeFileSync("decoded.png", Buffer.from(await res.arrayBuffer()));
})();
curl 
BASE=http://img-convert.cs.icbc
curl -sS -H "Content-Type: application/json" \
  -d '{"dataUrl":"data:image/png;base64,iVBORw0KGgoAAAANS..."}' \
  "$BASE/api/base64/to-image" -o decoded.png
Python 
import requests

BASE = "http://img-convert.cs.icbc"
r = requests.post(
    f"{BASE}/api/base64/to-image",
    json={"dataUrl": "data:image/png;base64,iVBORw0KGgoAAAANS..."},
)
open("decoded.png", "wb").write(r.content)

POST /api/svg/to-png（JSON：svg；可选 width、height）

Node.js 
const BASE = "http://img-convert.cs.icbc";

(async () => {
  const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="64" height="64"><rect width="64" height="64" fill="#34d399"/></svg>`;
  const res = await fetch(BASE + "/api/svg/to-png", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ svg }),
  });
  require("fs").writeFileSync("out.png", Buffer.from(await res.arrayBuffer()));
})();
curl 
BASE=http://img-convert.cs.icbc
curl -sS -H "Content-Type: application/json" \
  --data-binary @- "$BASE/api/svg/to-png" -o out.png <<'EOF'
{"svg":"<svg xmlns=\"http://www.w3.org/2000/svg\" width=\"64\" height=\"64\"><rect width=\"64\" height=\"64\" fill=\"#34d399\"/></svg>"}
EOF
Python 
import requests

BASE = "http://img-convert.cs.icbc"
svg = """<svg xmlns="http://www.w3.org/2000/svg" width="64" height="64"><rect width="64" height="64" fill="#34d399"/></svg>"""

POST /api/image/to-svg-wrap（multipart：file；可选 width、height）

Node.js 
const BASE = "http://img-convert.cs.icbc";
const fs = require("fs");
const { Blob } = require("buffer");

(async () => {
  const buf = fs.readFileSync("./photo.jpg");
  const blob = new Blob([buf], { type: "image/jpeg" });
  const fd = new FormData();
  fd.append("file", blob, "photo.jpg");
  fd.append("width", "1024");
  fd.append("height", "768");
  const res = await fetch(BASE + "/api/image/to-svg-wrap", { method: "POST", body: fd });
  fs.writeFileSync("wrapped.svg", await res.text());
})();
curl 
BASE=http://img-convert.cs.icbc
curl -sS -F "file=@./photo.jpg" -F "width=1024" -F "height=768" \
  "$BASE/api/image/to-svg-wrap" -o wrapped.svg
Python 
import requests

BASE = "http://img-convert.cs.icbc"
with open("photo.jpg", "rb") as f:
    r = requests.post(
        f"{BASE}/api/image/to-svg-wrap",
        files={"file": f},
        data={"width": "1024", "height": "768"},
    )
open("wrapped.svg", "w", encoding="utf-8").write(r.text)
